  3 Math.RAD = Math.PI / 180;
  4 
  5 Math.cosA = function(angle) {
  6 	return Math.cos(angle * Math.RAD);
  7 };
  8 
  9 Math.sinA = function(angle) {
 10 	return Math.sin(angle * Math.RAD);
 11 };

 33 //TODO: refactor to SEF or Raphael.fn namespace
 34 function PiePie(config) {
 35 
 36 	//TODO: validate config and data with schema
 37 
 38 	var 	pieCenterX 		= config.width / 2
 39 		,	pieCenterY 		= config.height / 2
 40 		,	pieInnerRadius	= config.width * 0.2
 41 		,	pieOuterRadius	= config.width * 0.35
 42 		,	pieSpotRadius	= (pieOuterRadius + pieInnerRadius) * 0.53
 43 		,	data = {}
 44 		, 	label = ""
 45 		,	rotation = config.rotation ? config.rotation : 0
 46 		;
 47 	
 48 	var paper = Raphael(config.x, config.y, config.width, config.height);
 49 	
 50 	paper.rect(0, 0, config.width, config.height).attr(config.background);
 51 
 52 	paper.customAttributes.arc = function (startAngle, endAngle, innerRadius, outerRadius) {
 53 		var
 54 			outerStartX	= pieCenterX - outerRadius * Math.cosA(startAngle)
 55 		,	outerStartY = pieCenterY - outerRadius * Math.sinA(startAngle)
 56 		,	outerEndX 	= pieCenterX - outerRadius * Math.cosA(endAngle)
 57 		,	outerEndY 	= pieCenterY - outerRadius * Math.sinA(endAngle)
 58 		,	innerStartX	= pieCenterX - innerRadius * Math.cosA(startAngle)
 59 		,	innerStartY	= pieCenterY - innerRadius * Math.sinA(startAngle)
 60 		,	innerEndX 	= pieCenterX - innerRadius * Math.cosA(endAngle)
 61 		,	innerEndY 	= pieCenterY - innerRadius * Math.sinA(endAngle)
 62 		;
 63 
 64         var p = [
 65           "M", outerStartX, outerStartY,
 66           "A", outerRadius, outerRadius, 0, 0, 1, outerEndX, outerEndY,
 67           "L", innerEndX, innerEndY,
 68           "A", innerRadius, innerRadius, 0, 0, 0, innerStartX, innerStartY,
 69           "L", outerStartX, outerStartY,
 70           "Z"
 71         ];
 72 		return {path: p};
 73 	};
 74 
 75 	function drawSector(offset, percent, color, onAnimationEnd) {
 76 		var startAngle = 180 * offset + rotation
 77 		,   endAngle = 180 * (offset + percent) + rotation
 78 		,   tipAngle = (startAngle + endAngle) / 2
 79 		,   tipCenterX = pieCenterX - pieSpotRadius * Math.cosA(tipAngle)
 80 		,   tipCenterY = pieCenterY - pieSpotRadius * Math.sinA(tipAngle)
 81 		,   tipEndX = pieCenterX - 1.1 * pieOuterRadius * Math.cosA(tipAngle)
 82 		,   tipEndY = pieCenterY - 1.1 * pieOuterRadius * Math.sinA(tipAngle)
 83 		,	tipOrientation = Math.cosA(tipAngle) > 0
 84 		,	tipLineEndX = tipEndX + (tipOrientation ? - 30: + 30)
 85 		;
 86 		
 87 		console.log(rotation, tipAngle);
 88 		
 89 		if(DEBUG) {
 90 			paper.circle(arcStartX, arcStartY, 3).attr({fill: "#0E0", stroke: "none"});
 91 			paper.circle(arcEndX, arcEndY, 3).attr({fill: "#00E", stroke: "none"});
 92 			paper.circle(pieCenterX, pieCenterY, pieOuterRadius).attr({stroke: "#fff"});
 93 		}
 94 
 95 		var baseColor = Raphael.rgb2hsb(color)
 96 		,	spotColor = $.extend(baseColor)
 97 		;
 98 		spotColor.b *= 0.9;
 99 		
100 		var set = paper.set();
101 
102 		function drawTip(){
103 			set.push(paper.circle(tipCenterX, tipCenterY, 3).attr({fill: "#000", stroke: "none"}));
104 			set.push(paper.path([
105 	            "M", tipCenterX, tipCenterY,
106 	            "L", tipEndX, tipEndY,
107 	            "L", tipLineEndX, tipEndY
108 	        ]));
109 			set.push(paper.text(tipLineEndX + (tipOrientation ? 12 : -12), tipEndY - 10, percent * 100 + "%"));
110 		}
111 
112 		var outerArc = paper.path()
113 							.attr({arc: [startAngle, startAngle, pieSpotRadius, pieOuterRadius]})
114 							.attr({fill: baseColor, stroke: "none"});
115 
116 		var innerArc = paper.path()
117 							.attr({arc: [startAngle, startAngle, pieInnerRadius, pieSpotRadius]})
118 							.attr({fill: spotColor, stroke: "none"});
119 		
120 		var outerAnimation = Raphael.animation(
121 			{arc: [startAngle, endAngle, pieSpotRadius, pieOuterRadius]}, 1000, ">"
122 			, function(){
123 				onAnimationEnd();
124 				drawTip();
125 				set.push(this);
126 				this.mouseover(function() {
127 					set.stop().animate({transform: "s1.1 1.1 " + pieCenterX + " " + pieCenterY}, 500, "elastic");
128 				}).mouseout(function () {
129 		            set.stop().animate({transform: ""}, 500, "elastic");
130 		        });
131 		});
132 		
133 		outerArc.animate(outerAnimation);
134 		
135 		innerArc.animateWith(outerArc, outerAnimation,
136 			{arc: [startAngle, endAngle, pieInnerRadius, pieSpotRadius]}, 1000, ">"
137 			, function(){
138 				set.push(this);
139 				this.mouseover(function() {
140 					set.stop().animate({transform: "s1.1 1.1 " + pieCenterX + " " + pieCenterY}, 500, "elastic");
141 				}).mouseout(function () {
142 		            set.stop().animate({transform: ""}, 500, "elastic");
143 		        });
144 		});
145 	}
146 
147 	function drawLegend(){
148 		var x = pieCenterX - pieOuterRadius
149 		,	y = pieCenterY +  config.height * 0.1
150 		,	legendWidth = 0
151 		,	legendHeight = 0
152 		;
153 
154 		switch(true) {
155 			case (rotation > 89 && rotation < 180): {
156 				y = pieCenterY - data.length * 10;
157 				x = config.width * 0.1;
158 				break;
159 			}
160 			case (rotation > 179 && rotation < 270): {
161 				y = config.height * 0.1;
162 				break;
163 			}
164 			case (rotation > 269): {
165 				x = pieCenterX + config.width * 0.05;
166 				y = pieCenterY - data.length * 10;
167 				break;
168 			}
169 		}
170 
171 		var l = paper.text(x, y, label.toUpperCase()).attr({font: "16px Arial", "font-weight": "bold", fill: "#000", "text-anchor": "start"});
172 		l = l.getBBox();
173 		legendWidth= l.width;
174 
175 		$(data).each(function(idx, obj){
176 			y += 20;
177 			var c = Raphael.rgb2hsb(config.colors[idx]);
178 			c.b *= 0.5;
179 			c = Raphael.hsb2rgb(c);
180 			var f = "315-" + c + "-" + config.colors[idx];
181 			paper.rect(x, y, 20, 10).attr({fill: f, stroke: "none"});
182 
183 			l = paper.text(x + 24, y + 4, obj.title.toUpperCase()).attr({"text-anchor": "start"});
184 			l = l.getBBox();
185 			legendWidth = l.width + 24 > legendWidth ? l.width + 24 : legendWidth; 
186 		});
187 	}

189 	$.getJSON(config.dataURL, function(res) {
190 		//TODO: implement colors generator
191 		//TODO: throw exception on error
192 		var pieOffset = 0;
193 		var animationQueue = $({});
194 		data = res.data;
195 		label = res.label;
196 		drawLegend();
197 		$(data).each(function(idx, obj){
198 			//TODO: implement values normalization and sorting
199 			animationQueue.queue(function(next) {
200 				var percent = parseFloat(obj.value);
201 				drawSector(pieOffset, percent, config.colors[idx], next);
202 				pieOffset += percent;
203 			});
204 		});
205 		if(config.endLoad) {
206 			config.endLoad();
207 		}
208 	});
209 	
210 	if(DEBUG) {
211 	 	paper.circle(pieCenterX, pieCenterY, 3).attr({fill: "#E00", stroke: "none"});
212 	 	paper.circle(pieCenterX - pieOuterRadius, pieCenterY, 3).attr({fill: "#E00", stroke: "none"});
213 	 	paper.circle(pieCenterX	+ pieOuterRadius, pieCenterY, 3).attr({fill: "#E00", stroke: "none"});
214 	 	paper.circle(pieCenterX, pieCenterY + pieOuterRadius, 3).attr({fill: "#E00", stroke: "none"});
215 	 	paper.circle(pieCenterX, pieCenterY - pieOuterRadius, 3).attr({fill: "#E00", stroke: "none"});
216 	}
217 	
218 	return paper;
219 };
